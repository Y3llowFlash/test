function dateDiff(date1, date2) { // Get the number of milliseconds between the two dates. var diff = date2 - date1; // Convert the milliseconds to days, hours, minutes, and seconds. var days = Math.floor(diff / 86400000); var hours = Math.floor((diff % 86400000) / 3600000); var minutes = Math.floor((diff % 3600000) / 60000); var seconds = Math.floor((diff % 60000) / 1000); // Return the difference in days, hours, minutes, and seconds. return { days: days, hours: hours, minutes: minutes, seconds: seconds }; }
